7    import org.apache.commons.lang3.StringUtils; 
8     
9    import java.util.LinkedList; 
10   import java.util.List; 
11    
12   public class TableBuilder 
13   { 
14       List<String[]> rows = new LinkedList<String[]>(); 
15    
16       public void addRow(String... cols) 
17       { 
18           rows.add(cols); 
19       } 
20    
21       private int[] colWidths() 
22       { 
23           int cols = -1; 
24    
25           for(String[] row : rows) 
26               cols = Math.max(cols, row.length); 
27    
28           int[] widths = new int[cols]; 
29    
30           for(String[] row : rows) { 
31               for(int colNum = 0; colNum < row.length; colNum++) { 
32                   widths[colNum] = 
33                           Math.max( 
34                                   widths[colNum], 
35                                   StringUtils.length(row[colNum])); 
36               } 
37           } 
38    
39           return widths; 
40       } 
41    
42       @Override 
43       public String toString() 
44       { 
45           StringBuilder buf = new StringBuilder(); 
46    
47           int[] colWidths = colWidths(); 
48    
49           for(String[] row : rows) { 
50               for(int colNum = 0; colNum < row.length; colNum++) { 
51                   buf.append( 
52                           StringUtils.rightPad( 
53                                   StringUtils.defaultString( 
54                                           row[colNum]), colWidths[colNum])); 
55                   buf.append(' '); 
56               } 
57    
58               buf.append('\n'); 
59           } 
60    
61           return buf.toString(); 
62       }

64   }
